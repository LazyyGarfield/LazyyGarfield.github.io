MyTinySTL-6.1.6【algo.h】

主要介绍了对区间整体操作的函数

  1. 【random_shuffle】函数，对区间进行随机重排
  2. 【rotate】函数，对区间内的两个长度不同的部分进行互换
  3. 【is_permutation】函数，判断区间2是否为区间1的排列组合
  4. 【next_permutation】函数和【prev_permutation】函数，判断区间是否有下一个排列组合和是否有上一个排列组合，返回bool

random_shuffle

将区间内的元素次序随机重排，具体操作为产生一个随机数，使得不同位置的值进行交换

// random_shuffle
// 将[first, last)内的元素次序随机重排
// 重载版本使用一个产生随机数的函数对象 rand
/*****************************************************************************************/
template <class RandomIter>
void random_shuffle(RandomIter first, RandomIter last)
{
  if (first == last)
    return;
  srand((unsigned)time(0));
  for (auto i = first + 1; i != last; ++i)
  {
    mystl::iter_swap(i, first + (rand() % (i - first + 1)));
  }
}

// 重载版本使用一个产生随机数的函数对象 rand
template <class RandomIter, class RandomNumberGenerator>
void random_shuffle(RandomIter first, RandomIter last,
                    RandomNumberGenerator& rand)
{
  if (first == last)
    return;
  auto len = mystl::distance(first, last);
  for (auto i = first + 1; i != last; ++i)
  {
    mystl::iter_swap(i, first + (rand(i - first + 1) % len));
  }
}

rotate

将[first, middle)内的元素和 [middle, last)内的元素互换，可以交换两个长度不同的区间，返回交换后middle的位置

第一个版本，forward_iterator_tag版本

template<class ForwardIter>
ForwardIter rotate_dispatch(ForwardIter first,ForwardIter middle,ForwardIter last,forward_iterator_tag){
    auto first2=middle;
    do{
        mystl::swap(*first,*first2);
        first++;first2++;
        if(first==middle){middle=first2;}
    }while(first2!=last); // 后半段移到前面
    
    auto new_middle=first;
    // 如果第一个区间比较短，此时new_middle则为返回的middle
    // 如果第一个区间比较长，则还需要继续调整
    first2=middle;
    while(first2!=last){
        mystl::swap(*first,*first2);
        *first++;*first2++;
        if(first==middle){middle=first2;} // 剩余调整区间比区间2短，则更新区间1
        else if(first2==last){first2=middle;} // 剩余调整区间比区间2长，则更新区间2
    }
    return new_middle;
}

第二个版本，bidirectional_iterator_tag版本

  1. 先将区间1和区间2都反转过来
  2. 区间1往后，区间2往前，不断交换其中的数值
  3. 当区间1到达末尾或者区间2到达头部时，此时将另一个区间的剩余部分进行反转即可
template<class BidirectionalIter>
BidirectionalIter rotate_dispatch(BidirectionalIter first,BidirectionalIter middle,BidirectionalIter last,bidirectional_iterator_tag){
    mystl::reverse_dispatch(first,middle,bidirectional_iterator_tag());
    mystl::reverse_dispatch(middle,last,bidirectional_iterator_tag());
    while(first!=middle && middle!=last){
        mystl::swap(*first++,*--last);
        if(first==middle){
            mystl::reverse_dispatch(middle,last,bidirectional_iterator_tag());
            return last;
        }
        else if(middle==last){
            mystl::reverse_dispatch(first,middle,bidirectional_iterator_tag());
            return first;
        }
    }
}

第三个版本，random_iterator_tag版本

// rotate_dispatch 的 random_access_iterator_tag 版本
template <class RandomIter>
RandomIter
rotate_dispatch(RandomIter first, RandomIter middle,
                RandomIter last, random_access_iterator_tag)
{
  // 因为是 random access iterator，我们可以确定每个元素的位置
  auto n = last - first;
  auto l = middle - first;
  auto r = n - l;
  // 先计算得到返回的迭代器
  auto result = first + (last - middle);
  if (l == r)
  {
    mystl::swap_ranges(first, middle, middle);
    return result;
  }
  // 求公因数
  auto cycle_times = rgcd(n, l);
  for (auto i = 0; i < cycle_times; ++i)
  {
    auto tmp = *first;
    auto p = first;
    // 区间1较短，不断来回交替swap
    if (l < r)
    {
      for (auto j = 0; j < r / cycle_times; ++j)
      {
        if (p > first + r)
        {
          *p = *(p - r);
          p -= r;
        }
        *p = *(p + l);
        p += l;
      }
    }
    // 区间2较短，不断来回交替swap
    else
    {
      for (auto j = 0; j < l / cycle_times - 1; ++j)
      {
        if (p < last - l)
        {
          *p = *(p + l);
          p += l;
        }
        *p = *(p - r);
        p -= r;
      }
    }
    *p = tmp;
    ++first;
  }
  return result;
}

其中，求公因数的函数为

template<class EuclideanRingElement>
EuclideanRingElement rgcd(EuclideanRingElement m,EuclideanRingElement n){
    while(n!=0){
        auto t=m%n;
        m=n;
        n=t;
    }
    return m;
}

将rotate整合起来

template <class ForwardIter>
ForwardIter
rotate(ForwardIter first, ForwardIter middle, ForwardIter last)
{
  if (first == middle)
    return last;
  if (middle == last)
    return first;
  return mystl::rotate_dispatch(first, middle, last, iterator_category(first));
}

【rotate_copy】函数，与【copy】行为类似，不同的是将结果复制到result所指的容器中，但是实现更加简单

  1. 先将区间2复制到result中
  2. 再将区间1复制到result中即可
template<class ForwardIter,class OutputIter>
OutputIter rotate_copy(ForwardIter first,ForwardIter middle,ForwardIter last,OutputIter result){
    return mystl::copy(first,middle,mystl::copy(middle,last,result));
}

is_permutation

判断[first1,last1)是否为[first2, last2)的排列组合

  1. 首先，如果迭代器是RandomIter，则可以计算两个区间的长度是否相等，如果不相等，则直接返回false

  2. 先找出相同的前缀段，无需考虑这部分

  3. 此时对剩余部分先进行判断：

    1）若迭代器类型是RandomIter，判断此时的区间是否达到终点，如果是，则直接返回true

    2）如果不是RandomIter，则判断此时两个区间的长度是否都为0，如果都为0，则返回true；如果长度不相等，则返回false

  4. 接下来，处理两个区间的剩余部分：

  5. 在区间1中遍历字段，记录从0到当前位置是否有数值和当前数值重复

    1）如果没有重复，则计算当前数值在整个区间2的数量和在整个区间1的数量，判断二者是否相同

    2）如果有重复，则说明第一次遇到该数值时，已经计算过，无需再次计算

// is_permutation
// 判断[first1,last1)是否为[first2, last2)的排列组合
/*****************************************************************************************/
template <class ForwardIter1, class ForwardIter2, class BinaryPred>
bool is_permutation_aux(ForwardIter1 first1, ForwardIter1 last1,
                        ForwardIter2 first2, ForwardIter2 last2,
                        BinaryPred pred)
{
  constexpr bool is_ra_it = mystl::is_random_access_iterator<ForwardIter1>::value
    && mystl::is_random_access_iterator<ForwardIter2>::value;
  if (is_ra_it)
  {
    auto len1 = last1 - first1;
    auto len2 = last2 - first2;
    if (len1 != len2)
      return false;
  }

  // 先找出相同的前缀段
  for (; first1 != last1 && first2 != last2; ++first1, (void) ++first2)
  {
    if (!pred(*first1, *first2))
      break;
  }
  if (is_ra_it)
  {
    if (first1 == last1)
      return true;
  }
  else
  {
    auto len1 = mystl::distance(first1, last1);
    auto len2 = mystl::distance(first2, last2);
    if (len1 == 0 && len2 == 0)
      return true;
    if (len1 != len2)
      return false;
  }

  // 判断剩余部分
  for (auto i = first1; i != last1; ++i)
  {
    bool is_repeated = false;
    for (auto j = first1; j != i; ++j)
    {
      if (pred(*j, *i))
      {
        is_repeated = true;
        break;
      }
    }

    if (!is_repeated)
    {
      // 计算 *i 在 [first2, last2) 的数目
      auto c2 = 0;
      for (auto j = first2; j != last2; ++j)
      {
        if (pred(*i, *j))
          ++c2;
      }
      if (c2 == 0)
        return false;

      // 计算 *i 在 [first1, last1) 的数目
      auto c1 = 1;
      auto j = i;
      for (++j; j != last1; ++j)
      {
        if (pred(*i, *j))
          ++c1;
      }
      if (c1 != c2)
        return false;
    }
  }
  return true;
}

整合起来

template <class ForwardIter1, class ForwardIter2, class BinaryPred>
bool is_permutation(ForwardIter1 first1, ForwardIter1 last1,
                    ForwardIter2 first2, ForwardIter2 last2,
                    BinaryPred pred)
{
  return is_permutation_aux(first1, last1, first2, last2, pred);
}


template <class ForwardIter1, class ForwardIter2>
bool is_permutation(ForwardIter1 first1, ForwardIter1 last1,
                    ForwardIter2 first2, ForwardIter2 last2)
{
  typedef typename iterator_traits<ForwardIter1>::value_type v1;
  typedef typename iterator_traits<ForwardIter2>::value_type v2;
  static_assert(std::is_same<v1, v2>::value,
                "the type should be same in mystl::is_permutation");
  return is_permutation_aux(first1, last1, first2, last2,
                            mystl::equal_to<v1>());
}

next_permuation

取得[first, last)所标示序列的下一个排列组合，如果没有下一个排序组合，返回 false，否则返回 true

下一个排列组合的意思是，【1354】的下一个排列组合是【1435】

  1. 首先判断是否为空或者是否只有1个元素，如果是则返回false
  2. 如果多于1个元素，则找到第一对小于关系的元素，如【1354】的【35】
  3. 然后在最后往前找到第一个大于【3】的元素【4】与【3】交换，交换之后后续元素全部翻转，最终为【1435】
  4. 如果最终不存在第一对小于关系的元素，比如【54321】，则将该区间翻转，并返回false
// next_permutation
// 取得[first, last)所标示序列的下一个排列组合，如果没有下一个排序组合，返回 false，否则返回 true
/*****************************************************************************************/
template <class BidirectionalIter>
bool next_permutation(BidirectionalIter first, BidirectionalIter last)
{
  auto i = last;
  if (first == last || first == --i)
    return false;
  for (;;)
  {
    auto ii = i;
    if (*--i < *ii)
    {                 // 找到第一对小于关系的元素
      auto j = last;
      while (!(*i < *--j)) {}
      mystl::iter_swap(i, j);       // 交换 i，j 所指元素
      mystl::reverse(ii, last);     // 将 ii 之后的所有元素反转
      return true;
    }
    if (i == first)
    {
      mystl::reverse(first, last);
      return false;
    }
  }
}

使用comp函数代替【<】的版本

// 重载版本使用函数对象 comp 代替比较操作
template <class BidirectionalIter, class Compared>
bool next_permutation(BidirectionalIter first, BidirectionalIter last, Compared comp)
{
  auto i = last;
  if (first == last || first == --i)
    return false;
  for (;;)
  {
    auto ii = i;
    if (comp(*--i, *ii))
    {
      auto j = last;
      while (!comp(*i, *--j)) {}
      mystl::iter_swap(i, j);       // 交换 i，j 所指元素
      mystl::reverse(ii, last);     // 将 ii 之后的所有元素反转
      return true;
    }
    if (i == first)
    {
      mystl::reverse(first, last);
      return false;
    }
  }
}

prev_permuation

取得[first, last)所标示序列的上一个排列组合，如果没有上一个排序组合，返回 false，否则返回 true

同理，上一个排列组合意思为，【1435】的上一个排列组合为【1354】

本质相同，先找到第一对大于关系的元素，然后从后往前找到一个小于当前值的元素进行交换，然后翻转后续部分

// prev_permutation
// 取得[first, last)所标示序列的上一个排列组合，如果没有上一个排序组合，返回 false，否则返回 true
/*****************************************************************************************/
template <class BidirectionalIter>
bool prev_permutation(BidirectionalIter first, BidirectionalIter last)
{
  auto i = last;
  if (first == last || first == --i)
    return false;
  for (;;)
  {
    auto ii = i;
    if (*ii < *--i)
    {                 // 找到第一对大于关系的元素
      auto j = last;
      while (!(*--j < *i)) {}
      mystl::iter_swap(i, j);       // 交换i，j
      mystl::reverse(ii, last);     // 将 ii 之后的所有元素反转
      return true;
    }
    if (i == first)
    {
      mystl::reverse(first, last);
      return false;
    }
  }
}

// 重载版本使用函数对象 comp 代替比较操作
template <class BidirectionalIter, class Compared>
bool prev_permutation(BidirectionalIter first, BidirectionalIter last, Compared comp)
{

  if (first == last || first == --i)
    return false;
  for (;;)
  {
    auto ii = i;
    if (comp(*ii, *--i))
    {
      auto j = last;
      while (!comp(*--j, *i)) {}
      mystl::iter_swap(i, j);       // 交换i，j
      mystl::reverse(ii, last);     // 将 ii 之后的所有元素反转
      return true;
    }
    if (i == first)
    {
      mystl::reverse(first, last);
      return false;
    }
  }
}
