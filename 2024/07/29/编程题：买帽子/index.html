编程题：买帽子

题目

买帽子

度度熊想去商场买一顶帽子，商场里有N顶帽子，有些帽子的价格可能相同。

度度熊想买一顶价格第三便宜的帽子，问第三便宜的帽子价格是多少？

描述

输入描述

  1. 首先输入一个正整数N（N <= 50）
  2. 第二行输入N个数表示每顶帽子的价格（价格均是正整数，且小于等于1000）

输出描述

  1. 如果存在第三便宜的帽子，请输出这个价格是多少，否则输出-1

示例1

输入

10
10 10 10 10 20 20 30 30 40 40

输出

30

示例2

输入

1
10

输出

-1

解答

  1. 基于堆排序即可解决
  2. 但是，该数组可能出现重复数字，因此需要另外定义一个哈希表去重
  3. 并且，当数字长度小于3时，返回-1
  4. 代码如下
#include <iostream>
#include <unordered_set>
#include <vector>
using namespace std;

void heapsort(vector<int> &nums,int i,int len){
    int idxmin=i,left=2*i+1,right=2*i+2;
    if(left<len&&nums[left]<nums[idxmin]){idxmin=left;}
    if(right<len&&nums[right]<nums[idxmin]){idxmin=right;}
    if(i!=idxmin){
        swap(nums[i],nums[idxmin]);
        heapsort(nums,idxmin,len);
    }
}

void heapbuild(vector<int> &nums){
    int n=nums.size();
    for(int i=n/2-1;i>=0;i--){
        heapsort(nums,i,n);
    }
}

int main() {
    // 初始化
    int n;
    cin>>n;
    // 使用哈希表去重
    unordered_set<int> s;
    vector<int> nums;
    for(int i=0;i<n;i++){
        int num;
        cin>>num;
        if(s.find(num)==s.end()){
            s.insert(num);
            nums.push_back(num);
        }
    }
    // 建立一个堆
    heapbuild(nums);
    int len=nums.size();
    // 如果长度不超过3，则返回-1
    if(len<3){
        cout<<-1<<endl;
        return 0;
    }
    // 进行两次弹出堆顶操作，可以找到第三小的数
    int k=3;
    for(int i=1;i<k;i++){
        swap(nums[0],nums[len-1]);
        len--;
        heapsort(nums,0,len);
    }
    cout<<nums[0]<<endl;
    return 0;
}
// 64 位输出请用 printf("%lld")
