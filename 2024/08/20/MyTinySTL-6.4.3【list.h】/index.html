6.4.3【list.h】【基础函数】

6.4.3节介绍【list】的功能和操作

赋值：assign

多种方式进行赋值

  1. 赋值为【n】个【value】
  2. 利用区间进行赋值
  void     assign(size_type n, const value_type& value) 
  { fill_assign(n, value); }

  template <class Iter, typename std::enable_if<
    mystl::is_input_iterator<Iter>::value, int>::type = 0>
  void     assign(Iter first, Iter last)
  { copy_assign(first, last); }

  void     assign(std::initializer_list<T> ilist)
  { copy_assign(ilist.begin(), ilist.end()); }

辅助函数【fill_assign】如下

  1. 如果当前空间不足n个元素，则利用【insert】函数插入
  2. 如果当前空间超出n个元素，则利用【erase】函数删除
// 用 n 个元素为容器赋值
template <class T>
void list<T>::fill_assign(size_type n, const value_type& value)
{
  auto i = begin();
  auto e = end();
  for (; n > 0 && i != e; --n, ++i)
  {
    *i = value;
  }
  if (n > 0)
  {
    insert(e, n, value);
  }
  else
  {
    erase(i, e);
  }
}

辅助函数【copy_assign】如下，与【fill_assign】函数类似

// 复制[f2, l2)为容器赋值
template <class T>
template <class Iter>
void list<T>::copy_assign(Iter f2, Iter l2)
{
  auto f1 = begin();
  auto l1 = end();
  for (; f1 != l1 && f2 != l2; ++f1, ++f2)
  {
    *f1 = *f2;
  }
  if (f2 == l2)
  {
    erase(f1, l1);
  }
  else
  {
    insert(l1, f2, l2);
  }
}

插入：emplace

【emplace_front】函数，创建一个【link_node】，然后利用【link_nodes_at_front】函数将该节点插入到链表的头部

  template <class ...Args>
  void     emplace_front(Args&& ...args)
  {
    THROW_LENGTH_ERROR_IF(size_ > max_size() - 1, "list<T>'s size too big");
    auto link_node = create_node(mystl::forward<Args>(args)...);
    link_nodes_at_front(link_node->as_base(), link_node->as_base());
    ++size_;
  }

辅助函数【link_nodes_at_front】，传入形参为代表【first,last】的两个【base_ptr】

// 在头部连接 [first, last] 结点
template <class T>
void list<T>::link_nodes_at_front(base_ptr first, base_ptr last)
{
  first->prev = node_;
  last->next = node_->next;
  last->next->prev = last;
  node_->next = first;
}

【emplace_back】函数，与【emplace_front】函数同理，利用【link_nodes_at_back】函数

  template <class ...Args>
  void     emplace_back(Args&& ...args)
  {
    THROW_LENGTH_ERROR_IF(size_ > max_size() - 1, "list<T>'s size too big");
    auto link_node = create_node(mystl::forward<Args>(args)...);
    link_nodes_at_back(link_node->as_base(), link_node->as_base());
    ++size_;
  }

【emplace】函数，在【pos】处插入元素

  template <class ...Args>
  iterator emplace(const_iterator pos, Args&& ...args)
  {
    THROW_LENGTH_ERROR_IF(size_ > max_size() - 1, "list<T>'s size too big");
    auto link_node = create_node(mystl::forward<Args>(args)...);
    link_nodes(pos.node_, link_node->as_base(), link_node->as_base());
    ++size_;
    return iterator(link_node);
  }

辅助函数【link_nodes】如下

// 在 pos 处连接 [first, last] 的结点
template <class T>
void list<T>::link_nodes(base_ptr pos, base_ptr first, base_ptr last)
{
  pos->prev->next = first;
  first->prev = pos->prev;
  pos->prev = last;
  last->next = pos;
}

插入：insert

插入1个【value】的版本，分为左值引用和右值引用

  iterator insert(const_iterator pos, const value_type& value)
  {
    THROW_LENGTH_ERROR_IF(size_ > max_size() - 1, "list<T>'s size too big");
    auto link_node = create_node(value);
    ++size_;
    return link_iter_node(pos, link_node->as_base());
  }

  iterator insert(const_iterator pos, value_type&& value)
  {
    THROW_LENGTH_ERROR_IF(size_ > max_size() - 1, "list<T>'s size too big");
    auto link_node = create_node(mystl::move(value));
    ++size_;
    return link_iter_node(pos, link_node->as_base());
  }

其中辅助函数【link_iter_node】如下，利用之前定义的辅助函数即可完成

// 在 pos 处连接一个节点
template <class T>
typename list<T>::iterator 
list<T>::link_iter_node(const_iterator pos, base_ptr link_node)
{
  if (pos == node_->next)
  {
    link_nodes_at_front(link_node, link_node);
  }
  else if (pos == node_)
  {
    link_nodes_at_back(link_node, link_node);
  }
  else
  {
    link_nodes(pos.node_, link_node, link_node);
  }
  return iterator(link_node);
}

插入n个元素：insert

在【pos】处插入【n】个【value】

  iterator insert(const_iterator pos, size_type n, const value_type& value)
  { 
    THROW_LENGTH_ERROR_IF(size_ > max_size() - n, "list<T>'s size too big");
    return fill_insert(pos, n, value); 
  }

其中辅助函数【fill_insert】如下，具体实现为：

  1. 先创建一个节点和一个区间（由表示首尾的两个迭代器组成），然后将剩余n-1个节点依次加入到该区间中
  2. 利用【link_nodes】函数将上述区间插入到【pos】位置
// 在 pos 处插入 n 个元素
template <class T>
typename list<T>::iterator 
list<T>::fill_insert(const_iterator pos, size_type n, const value_type& value)
{
  iterator r(pos.node_);
  if (n != 0)
  {
    const auto add_size = n;
    auto node = create_node(value);
    node->prev = nullptr;
    r = iterator(node);
    iterator end = r;
    try
    {
      // 前面已经创建了一个节点，还需 n - 1 个
      for (--n; n > 0; --n, ++end)
      {
        auto next = create_node(value);
        end.node_->next = next->as_base();  // link node
        next->prev = end.node_;
      }
      size_ += add_size;
    }
    catch (...)
    {
      auto enode = end.node_;
      while (true)
      {
        auto prev = enode->prev;
        destroy_node(enode->as_node());
        if (prev == nullptr)
          break;
        enode = prev;
      }
      throw;
    }
    link_nodes(pos.node_, r.node_, end.node_);
  }
  return r;
}

插入一段区间：insert

  template <class Iter, typename std::enable_if<
    mystl::is_input_iterator<Iter>::value, int>::type = 0>
  iterator insert(const_iterator pos, Iter first, Iter last)
  { 
    size_type n = mystl::distance(first, last);
    THROW_LENGTH_ERROR_IF(size_ > max_size() - n, "list<T>'s size too big");
    return copy_insert(pos, n, first); 
  }

其中辅助函数【copy_insert】如下，与【fill_insert】函数类似

// 在 pos 处插入 [first, last) 的元素
template <class T>
template <class Iter>
typename list<T>::iterator 
list<T>::copy_insert(const_iterator pos, size_type n, Iter first)
{
  iterator r(pos.node_);
  if (n != 0)
  {
    const auto add_size = n;
    auto node = create_node(*first);
    node->prev = nullptr;
    r = iterator(node);
    iterator end = r;
    try
    {
      for (--n, ++first; n > 0; --n, ++first, ++end)
      {
        auto next = create_node(*first);
        end.node_->next = next->as_base();  // link node
        next->prev = end.node_;
      }
      size_ += add_size;
    }
    catch (...)
    {
      auto enode = end.node_;
      while (true)
      {
        auto prev = enode->prev;
        destroy_node(enode->as_node());
        if (prev == nullptr)
          break;
        enode = prev;
      }
      throw;
    }
    link_nodes(pos.node_, r.node_, end.node_);
  }
  return r;
}

弹出：pop

【pop_front】函数，删去头节点，因此对【node_->next】操作

【pop_back】函数，删去尾节点，因此对【node_->prev】操作

  // pop_front / pop_back

  void pop_front() 
  {
    MYSTL_DEBUG(!empty());
    auto n = node_->next;
    unlink_nodes(n, n);
    destroy_node(n->as_node());
    --size_;
  }

  void pop_back() 
  { 
    MYSTL_DEBUG(!empty());
    auto n = node_->prev;
    unlink_nodes(n, n);
    destroy_node(n->as_node());
    --size_;
  }

其中所用到的辅助函数【unlink_nodes】如下，使容器与【first,last】断开连接

// 容器与 [first, last] 结点断开连接
template <class T>
void list<T>::unlink_nodes(base_ptr first, base_ptr last)
{
  first->prev->next = last->next;
  last->next->prev = first->prev;
}

删除：erase

第一个版本，删除【pos】位置上的值，返回下一个位置的迭代器

// 删除 pos 处的元素
template <class T>
typename list<T>::iterator 
list<T>::erase(const_iterator pos)
{
  MYSTL_DEBUG(pos != cend());
  auto n = pos.node_;
  auto next = n->next;
  unlink_nodes(n, n);
  destroy_node(n->as_node());
  --size_;
  return iterator(next);
}

第二个版本，删除区间【first,last】内的值，需要注意的是，区间内每个值都需要【destroy】

// 删除 [first, last) 内的元素
template <class T>
typename list<T>::iterator 
list<T>::erase(const_iterator first, const_iterator last)
{
  if (first != last)
  {
    unlink_nodes(first.node_, last.node_->prev);
    while (first != last)
    {
      auto cur = first.node_;
      ++first;
      destroy_node(cur->as_node());
      --size_;
    }
  }
  return iterator(last.node_);
}

清空：clear

具体实现如下，先从头节点【node_->next】开始，依次【destroy】每个值，然后最后到达尾节点时，利用【unlink】函数释放尾节点的连接，不需要对尾节点进行【destroy】

// 清空 list
template <class T>
void list<T>::clear()
{
  if (size_ != 0)
  {
    auto cur = node_->next;
    for (base_ptr next = cur->next; cur != node_; cur = next, next = cur->next)
    {
      destroy_node(cur->as_node());
    }
    node_->unlink();
    size_ = 0;
  }
}

重置容量：resize

将容器重置容量为【_Newsize】，如果当前容量小于【_Newsize】，则超出区间均赋值为【value】

// 重置容器大小
template <class T>
void list<T>::resize(size_type new_size, const value_type& value)
{
  auto i = begin();
  size_type len = 0;
  while (i != end() && len < new_size)
  {
    ++i;
    ++len;
  }
  if (len == new_size)
  {
    erase(i, node_);
  }
  else
  {
    insert(node_, new_size - len, value);
  }
}
