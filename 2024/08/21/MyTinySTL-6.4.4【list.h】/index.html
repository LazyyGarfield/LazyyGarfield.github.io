6.4.4【list.h】【相关函数】

6.4.4节介绍【list】的更复杂的功能和操作

接合：splice

第一个版本，将【list& x】接合于【pos】之前，具体实现为

  1. 利用【unlink_nodes】函数将【list& x】的首尾断开连接
  2. 利用【link_nodes】函数将【list& x】接在【pos.node_】处
  3. 需要注意的是，要调整容器和【list& x】的size大小
// 将 list x 接合于 pos 之前
template <class T>
void list<T>::splice(const_iterator pos, list& x)
{
  MYSTL_DEBUG(this != &x);
  if (!x.empty())
  {
    THROW_LENGTH_ERROR_IF(size_ > max_size() - x.size_, "list<T>'s size too big");

    auto f = x.node_->next;
    auto l = x.node_->prev;

    x.unlink_nodes(f, l);
    link_nodes(pos.node_, f, l);

    size_ += x.size_;
    x.size_ = 0;
  }
}

第二个版本，将【list& x】中【const_iterator it】所指的节点接合到【pos】之前，与第一个版本相比，只接合了一个节点

// 将 it 所指的节点接合于 pos 之前
template <class T>
void list<T>::splice(const_iterator pos, list& x, const_iterator it)
{
  if (pos.node_ != it.node_ && pos.node_ != it.node_->next)
  {
    THROW_LENGTH_ERROR_IF(size_ > max_size() - 1, "list<T>'s size too big");

    auto f = it.node_;

    x.unlink_nodes(f, f);
    link_nodes(pos.node_, f, f);

    ++size_;
    --x.size_;
  }
}

第三个版本，将【list& x】中的【first,last】内的节点接合到【pos】之前

// 将 list x 的 [first, last) 内的节点接合于 pos 之前
template <class T>
void list<T>::splice(const_iterator pos, list& x, const_iterator first, const_iterator last)
{
  if (first != last && this != &x)
  {
    size_type n = mystl::distance(first, last);
    THROW_LENGTH_ERROR_IF(size_ > max_size() - n, "list<T>'s size too big");
    auto f = first.node_;
    auto l = last.node_->prev;

    x.unlink_nodes(f, l);
    link_nodes(pos.node_, f, l);

    size_ += n;
    x.size_ -= n;
  }
}

移除：remove

第一个版本，移除掉值为【value】的节点

  void remove(const value_type& value)
  { remove_if([&](const value_type& v) {return v == value; }); }

第二个版本，移除掉满足一元条件的节点

// 将另一元操作 pred 为 true 的所有元素移除
template <class T>
template <class UnaryPredicate>
void list<T>::remove_if(UnaryPredicate pred)
{
  auto f = begin();
  auto l = end();
  for (auto next = f; f != l; f = next)
  {
    ++next;
    if (pred(*f))
    {
      erase(f);
    }
  }
}

去重：unique

移除两个相邻元素满足二元条件中的后一个元素

  void unique()
  { unique(mystl::equal_to<T>()); }
  template <class BinaryPredicate>
  void unique(BinaryPredicate pred);


// 移除 list 中满足 pred 为 true 重复元素
template <class T>
template <class BinaryPredicate>
void list<T>::unique(BinaryPredicate pred)
{
  auto i = begin();
  auto e = end();
  auto j = i;
  ++j;
  while (j != e)
  {
    if (pred(*i, *j))
    {
      erase(j);
    }
    else
    {
      i = j;
    }
    j = i;
    ++j;
  }
}

合并：merge

第一个版本，按升序排列将容器和【list& x】合并起来

  void merge(list& x)
  { merge(x, mystl::less<T>()); }

第二个版本，按自定义的【Compare comp】排列将容器和【list& x】合并起来，具体实现为：

  1. 定义两个指针【f1】、【f2】分别为容器和【list& x】的当前节点，比较当前两个节点满足条件
  2. 如果满足条件的话，则【f2】继续往下判断，直到不满足条件，此时将该区间内的节点都插入到容器中，然后【f1++】，继续判断节点是否满足条件
  3. 如果不满足条件的话，则【f1++】，继续判断节点是否满足条件
  4. 直到【f1】到达容器的末尾或者【f2】到达【list& x】的末尾
  5. 如果【list& x】还有剩余，则将其插入到容器的尾部
// 与另一个 list 合并，按照 comp 为 true 的顺序
template <class T>
template <class Compare>
void list<T>::merge(list& x, Compare comp)
{
  if (this != &x)
  {
    THROW_LENGTH_ERROR_IF(size_ > max_size() - x.size_, "list<T>'s size too big");

    auto f1 = begin();
    auto l1 = end();
    auto f2 = x.begin();
    auto l2 = x.end();

    while (f1 != l1 && f2 != l2)
    {
      if (comp(*f2, *f1))
      {
        // 使 comp 为 true 的一段区间
        auto next = f2;
        ++next;
        for (; next != l2 && comp(*next, *f1); ++next)
          ;
        auto f = f2.node_;
        auto l = next.node_->prev;
        f2 = next;

        // link node
        x.unlink_nodes(f, l);
        link_nodes(f1.node_, f, l);
        ++f1;
      }
      else
      {
        ++f1;
      }
    }
    // 连接剩余部分
    if (f2 != l2)
    {
      auto f = f2.node_;
      auto l = l2.node_->prev;
      x.unlink_nodes(f, l);
      link_nodes(l1.node_, f, l);
    }

    size_ += x.size_;
    x.size_ = 0;
  }
}

排序：sort

利用【list_sort】函数，传入参数分别为表示区间首尾的两个迭代器，区间的大小以及排序的方式【Compared comp】

  void sort()
  { list_sort(begin(), end(), size(), mystl::less<T>()); }
  template <class Compared>
  void sort(Compared comp)
  { list_sort(begin(), end(), size(), comp); }

【list_sort】函数如下（基于归并排序思想）

  1. 先讨论区间大小是否为1，如果为1的话直接返回【f1】
  2. 讨论区间大小是否为2，如果为2的话直接比较【f1】和【–l2】，然后进行排序
  3. 当区间长度大于2时，将区间切分为两部分，递归使用【list_sort】函数
  4. 对两部分进行合并，即【merge】算法
// 对 list 进行归并排序，返回一个迭代器指向区间最小元素的位置
template <class T>
template <class Compared>
typename list<T>::iterator 
list<T>::list_sort(iterator f1, iterator l2, size_type n, Compared comp)
{
  if (n < 2)
    return f1;

  if (n == 2)
  {
    if (comp(*--l2, *f1))
    {
      auto ln = l2.node_;
      unlink_nodes(ln, ln);
      link_nodes(f1.node_, ln, ln);
      return l2;
    }
    return f1;
  }

  auto n2 = n / 2;
  auto l1 = f1;
  mystl::advance(l1, n2);
  auto result = f1 = list_sort(f1, l1, n2, comp);  // 前半段的最小位置
  auto f2 = l1 = list_sort(l1, l2, n - n2, comp);  // 后半段的最小位置

  // 合并两段有序区间
  while (f1 != l1 && f2 != l2)
  {
    if (comp(*f2, *f1))
    {
      auto m = f2;
      ++m;
      for (; m != l2 && comp(*m, *f1); ++m)
        ;
      auto f = f2.node_;
      auto l = m.node_->prev;
      if (l1 == f2)
        l1 = m;
      f2 = m;
      unlink_nodes(f, l);
      m = f1;
      ++m;
      link_nodes(f1.node_, f, l);
      f1 = m;
    }
    else
    {
      ++f1;
    }
  }
  return result;
}

反转：reverse

将链表反转

// 将 list 反转
template <class T>
void list<T>::reverse()
{
  if (size_ <= 1)
  {
    return;
  }
  auto i = begin();
  auto e = end();
  while (i.node_ != e.node_)
  {
    mystl::swap(i.node_->prev, i.node_->next);
    i.node_ = i.node_->prev;
  }
  mystl::swap(e.node_->prev, e.node_->next);
}
